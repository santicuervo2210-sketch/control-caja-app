# Configuración de Supabase
# Obtén estos valores desde https://supabase.com/dashboard/project/[tu-proyecto]/settings/api

# URL de tu proyecto Supabase
NEXT_PUBLIC_SUPABASE_URL=https://tu-proyecto.supabase.co

# Clave anónima (public) de Supabase
NEXT_PUBLIC_SUPABASE_ANON_KEY=tu-anon-key-aqui

# Instrucciones para configurar Supabase:
# 1. Ve a https://supabase.com y crea una cuenta gratis
# 2. Crea un nuevo proyecto
# 3. Ve a Settings > API
# 4. Copia la URL del proyecto y la anon key
# 5. Crea un archivo .env.local en la raíz del proyecto
# 6. Agrega las variables de arriba con tus valores reales
# 7. Reinicia la aplicación
